### operators

```python
print(7//3)  # 整除
print(7%3)   # 求余数
print(7/3)   # 直接除
print(round(1.6), round(1.3)) # 四舍五入
import math
print(math.ceil(1.23))  # 向上取整
print(math.floor(1.6))  # 向下取整

1 + 2
_       # 下划线_存储了上一次运行的结果
_ + 1
_

x = 5  
5 == x > 4  # 链式比较，相当于x == 5 and x > 4  

print(0&1, 1&1, 2&1, 3&1)
print(3&5)
if num&1 == 0:  # 另一种判断偶数的方法
    print("num是偶数")
    

print(1|0, 0|0)

print(1^0, 1^1, 0^0)
print(3^5)

print(~1, ~0, ~5)
```
